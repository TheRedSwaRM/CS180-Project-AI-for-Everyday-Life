% Completely hide content
\newcommand{\hidden}[1]{%
    \hphantom{\vphantom{#1}}%
}

% Wide figure block \wfig[width]{filename, caption}
\newcommand{\wfig}[3][1]{\begin{figure}[H]%
    \centering%
    \includegraphics[width=#1\linewidth]{resources/#2}%
    \caption{\centering {#3}}%
    \label{fig:#2}%
\end{figure}}

% Code block environment
\DeclareFloatingEnvironment[name={Code Block}]{cb}
\crefname{cb}{Code Block}{Code Blocks}

% Full imported code block
\newcommand{\cbcode}[3]{\begin{cb}[H]%
    \centering%
    \inputminted[breaklines, linenos, xleftmargin=\dimexpr \paperwidth-\textwidth \relax, fontsize=\footnotesize]{#2}{resources/#1}%
    \caption{\centering {#3}}%
    \label{cb:#1}%
\end{cb}}

% Truncated code block
\newcommand{\cbcodet}[5]{\begin{cb}[H]%
    \centering%
    \inputminted[breaklines, linenos, xleftmargin=\dimexpr \paperwidth-\textwidth \relax, fontsize=\footnotesize, firstline=#3, lastline=#4]{#2}{resources/#1}%
    \caption{\centering {#5}}%
    \label{cb:#1}%
\end{cb}}
